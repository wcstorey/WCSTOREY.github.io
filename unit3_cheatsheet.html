CREATE/UPDATE/DELETE | FINDING DATA

CREATE/UPDATE/DELETE

Command

Description
CREATE TABLE "table_name" ("column_1" "data_type_for_column_1", "column_2" "data_type_for_column_2", ... ) creates a table named table_name with columns named column_1 and column_2, etc. requiring the data types specified for each column
ALTER TABLE table_name ADD column_name datatype Add a column requiring the specified data type in an existing table
ALTER TABLE table_name DDROP column_name datatype Delete a column in an existing table
DROP TABLE table_name Delete table called table_name

FINDING DATA

Command

Description
SELECT * FROM table returns all data from table
SELECT column_name FROM table returns data from column_name in table
SELECT DISTINCT column_name FROM table returns distinct data from column name (i.e., no same data is listed more than once in the returned data)
SELECT column_name FROM table WHERE column_name2=value returns from column_name data only values that
SELECT column_name FROM table WHERE column_name IN (value_1, value_2) returns from column_name data for only these values
SELECT column_name FROM table WHERE column_name LIKE "pattern%" returns from column_name data for values that only begin with this pattern
SELECT column_name FROM table WHERE column_name LIKE "%pattern" returns from column_name data for values that only end with this pattern
SELECT column_name FROM table WHERE column_name LIKE "%pattern%" returns from column_name data for values that have this pattern anywhere in the string
SELECT column_name FROM table WHERE column_name LIKE [a..f] returns from column_name data for values that only have this range of characters in their string
SELECT column_name FROM table WHERE statement AND statement returns from column_name data for values that satisfy both of these statements
